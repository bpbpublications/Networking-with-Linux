# Install the OpenSSL package
sudo apt install openssl

# Create a self signed certificate
sudo openssl req -x509 -nodes -days 365 -newkey rsa:2048 \
-keyout /etc/ssl/private/nginx-selfsigned.key \
-out /etc/ssl/certs/nginx-selfsigned.crt

# Update /etc/nginx/sites-available/example.com as follows
server { 
listen 80; 
server_name example.com www.example.com; 
root /var/www/example.com/html; 
	return 301 https://$host$request_uri;
 } 
server { 
listen 443 ssl;
server_name example.com www.example.com;
ssl_certificate /etc/ssl/certs/nginx-selfsigned.crt;
	ssl_certificate_key /etc/ssl/private/nginx-selfsigned.key;

	root /var/www/example.com/html;
	index index.html index.htm;

	location / { 
    		try_files $uri $uri/ =404; 
	} 
}

# Access the website with curl skipping certificate validation
curl -kL  http://example.com

# Add the self signed certificate to the trust store
sudo cp /etc/ssl/certs/nginx-selfsigned.crt /usr/local/share/ca-certificates/
sudo update-ca-certificates
