# The maximum number of remembered connection requests (SYN packets)
sysctl net.ipv4.tcp_max_syn_backlog

# Reduce SYN-ACK retries to clear half-open connections faster
sysctl net.ipv4.tcp_synack_retries  

# The maximum number of established connections, waiting to be accepted by the application
sysctl  net.core.somaxconn


# Tuning the values for your environment
sysctl –w net.ipv4.tcp_max_syn_backlog=65536
sysctl –w net.core.somaxconn=65536
