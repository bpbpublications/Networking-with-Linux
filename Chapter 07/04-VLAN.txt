# Create the VLAN sub-interface
sudo ip link add link eth0 name eth0.100 type vlan id 100

# Create a namespace and move the VLAN interface
sudo ip netns add vlan100-ns
sudo ip link set eth0.100 netns vlan100-ns

# Configure the interface in the namespace
sudo ip netns exec vlan100-ns ip link set lo up 
sudo ip netns exec vlan100-ns ip link set eth0.100 up
sudo ip netns exec vlan100-ns ip addr add 192.168.100.50/24 dev eth0.100
sudo ip netns exec vlan100-ns ip route add default via 192.168.100.1


# Verify the interface configuration and connectivity
sudo ip netns exec vlan100-ns ip addr show eth0.100
sudo ip netns exec vlan100-ns ping 192.168.100.1
